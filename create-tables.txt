
#CREATE TABLES:


CREATE TABLE Teams 
(team_name VARCHAR(50) PRIMARY KEY, captain VARCHAR(30));

CREATE TABLE Players 
(player_id INT PRIMARY KEY, player_name VARCHAR(30), team_name VARCHAR(50), role VARCHAR(50), UNIQUE(player_id, team_id));

CREATE TABLE Matches
(match_id INT PRIMARY KEY, team1_name VARCHAR(50), team2_name VARCHAR(50), match_date DATE,
 venue VARCHAR(64), winner_name VARCHAR(50), team1_runs INT, team2_runs INT, 
 team1_balls INT, team2_balls INT, team1_wickets INT, team2_wickets INT);

CREATE TABLE Scorecard
(match_id INT, player_id INT PRIMARY KEY, team_name VARCHAR(50), runs INT, balls INT, 
fours INT, sixes INT, wickets INT);
